class SudokuBoard {
    private int[][] board;
    private int[] availableNumbers;

    public SudokuBoard(int[][] init) {
        this.board = init;
        this.availableNumbers = new int[9];

        for (int i = 0; i < 9; i++) {
            this.availableNumbers[i] = 1;
        }
    }

    public void print() { // print the entire board by traversing the 2d array, elementary at this point
        System.out.println("Unsolved board:");

        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[i].length; j++) {
                System.out.print((board[i][j] + 1) + " ");
            }
            System.out.println();
        }

        System.out.println("\n");
    }

    private boolean noNumbersLeft() { // check if all numbers are used up
        for (int i = 0; i < availableNumbers.length; i++) {
            if (availableNumbers[i] != 0) {
                return false;
            }
        }
        return true;
    }

    private boolean isOk(int r, int c) { // check rows separately to ensure move is in line with rules
        for (int row = 0; row < board.length; row++) {
            if (board[row][c] == board[r][c] && row != r) {
                return false;
            }
        }

        for (int col = 0; col < board.length; col++) { // check columns separately to make sure it is within the rules
            if (board[r][col] == board[r][c] && col != c) {
                return false;
            }
        }

        for (int rTemp = (r / 3) * 3; rTemp < (r / 3) * 3 + 3; rTemp++) { // check the 3x3 square for duplicates,
            for (int cTemp = (c / 3) * 3; cTemp < (c / 3) * 3 + 3; cTemp++) {
                if ((board[rTemp][cTemp] == board[r][c]) && ((r != rTemp) && (c != (cTemp)))) { // return false if the current value being tested equals the value of the number just placed and is not at the same index
                    return false;
                }
            }
        }

        return true; // return true otherwise
    }

    public boolean solve() {
        if (noNumbersLeft()) {
            return true; // if no numbers are left, it is "solved" as much as possible
        } else {
            for (int row = 0; row < board.length; row++) {
                for (int col = 0; col < board[row].length; col++) {
                    if (board[row][col] == 0) { // check if current cell is empty
                        for (int num = 1; num <= 9; num++) { // try each number from 1 to 9
                            if (isOk(row, col)) { // check if the move is valid
                                board[row][col] = num; // set current cell to the valid number
                                availableNumbers[num - 1]--; // mark used number as unavailable
                                if (solve()) { // recursive call to solve the rest of the puzzle
                                    return true;
                                } else {
                                    board[row][col] = 0; // restore previous cell value
                                    availableNumbers[num - 1]++; // mark used number as available again
                                }
                            }
                        }
                    }
                }
            }
        }

        return false; // if no solution was found, return false
    }
}
